CREATE TABLE course (
    id varchar2(50),
    teacher_id varchar2(50) NOT NULL,
    name varchar2(100) NOT NULL,
    info varchar2(3000),
    code varchar2(100) NOT NULL,
    semester int NOT null,
    IS_active int CHECK(is_active BETWEEN 0 AND 1) NOT null,
    created_time DATE,
    PRIMARY KEY (id),
    FOREIGN KEY (id)
        REFERENCES std_course(id)
        ON DELETE CASCADE,
    FOREIGN KEY (teacher_id)
        REFERENCES teacher(id)
        ON DELETE CASCADE
)TABLESPACE SE_DATA;
