CREATE TABLE class (
    id int,
    course_id varchar2(50) NOT NULL,
    title varchar2(100) NOT NULL,
    description varchar2(3000),
    content_url varchar2(512),
    create_time DATE,
    PRIMARY KEY (id,course_id),
    FOREIGN KEY (course_id)
        REFERENCES course(id)
        ON DELETE CASCADE
)TABLESPACE SE_DATA;
